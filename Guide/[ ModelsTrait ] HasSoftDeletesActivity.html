# HasSoftDeletesActivity

  • provide eloquent functions working with softdelete traits.





 
 



 


<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\SoftDeletes;
use Pylon\Traits\Models\HasSoftDeletesActivity;

class Flight extends Model
{
    use SoftDeletes, HasSoftDeletesActivity;
}

# Available Methods

# whereActivity

Param Description
$value 0 - onlyTrashed(), 2 - withTrashed()
  • help return either onlyTrashed() or withTrashed().

Example




 

 




    $fight = Flight::query();

	$fight->whereActivity(0); // return ->onlyTrashed()

    $fight->whereActivity(2); // return ->withTrashed()

    // ...
