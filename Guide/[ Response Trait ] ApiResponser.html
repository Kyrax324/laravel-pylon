# ApiResponser

provide methods to standardizing the format of api response.

# Available Methods

# successResponse

Param Description Default
$message success message -
$result success result null
$code HTTP status code 200
  • help to return success api response
return $this->successResponse('Success',"Some Result");
1
// output
{
  "status": 200,
  "message": "Success",
  "result": "Some Result"
}

# errorResponse

Param Description Default
$message error message -
$error error type 'ERROR'
$code HTTP status code 400
  • help to return error api response
return $this->errorResponse('Some Error Message');
1
// output
{
  "status": 400,
  "message": "Success",
  "result": "Some Result"
