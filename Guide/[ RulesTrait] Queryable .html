# Example

 



 






 
 
 




use Pylon\Traits\Rules\Queryable;

class SomeRequest extends FormRequest
{
	use Queryable;

	public function rules()
	{
		return array_merge([
				// some rules
			],
			$this->pageables,
			$this->sortables,
			$this->searchables,
		);
	}
